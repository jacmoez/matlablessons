Matlab Array

In MATLAB, an array is a fundamental data structure used to store a collection of values or elements. Arrays can represent data in multiple dimensions, such as 1-D vectors (either row or column vectors), 2-D matrices, or even higher-dimensional arrays. MATLAB arrays are flexible and support different types of elements such as numbers, characters, or logical values.

% Example 1: Row vector
rowVec = [1, 2, 3, 4, 5];
disp('Row Vector Elements:');
for i = 1:length(rowVec)
  disp(rowVec(i));
end

% Example 2: Column vector
colVec = [1; 2; 3; 4; 5];
disp('Column Vector Elements:');
for i = 1:length(colVec)
  disp(colVec(i));
end

% Example 3: 2D matrix
A = [1 2 3; 4 5 6; 7 8 9];
disp('2D Matrix Elements:');
for i = 1:size(A, 1)
 for j = 1:size(A, 2)
  disp(A(i, j));
 end
end

% Example 4: 3D array
B = rand(2, 2, 2);
disp('3D Array Elements:');
for i = 1:size(B, 1)
  for j = 1:size(B, 2)
   for k = 1:size(B, 3)
    disp(B(i, j, k));
   end
  end
end

% Example 5: Linspace array
linArray = linspace(0, 10, 5);
disp('Linspace Array Elements:');
for i = 1:length(linArray)
  disp(linArray(i));
end

% Example 6: Logarithmic spaced array
logArray = logspace(1, 3, 5);
disp('Logarithmic Spaced Array Elements:');
for i = 1:length(logArray)
  disp(logArray(i));
end
% Example 7: Array of zeros
zeroArray = zeros(3, 4);
disp('Zero Array Elements:');
for i = 1:size(zeroArray, 1)
  for j = 1:size(zeroArray, 2)
   disp(zeroArray(i, j));
  end
end

% Example 8: Array of ones
oneArray = ones(4, 3);
disp('One Array Elements:');
for i = 1:size(oneArray, 1)
  for j = 1:size(oneArray, 2)
   disp(oneArray(i, j));
  end
end

% Example 9: Identity matrix
I = eye(4);
disp('Identity Matrix Elements:');
for i = 1:size(I, 1)
  for j = 1:size(I, 2)
   disp(I(i, j));
  end
end

% Example 10: Random numbers array
randArray = randn(3, 3);
disp('Random Numbers Array Elements:');
for i = 1:size(randArray, 1)
  for j = 1:size(randArray, 2)
   disp(randArray(i, j));
  end
end
